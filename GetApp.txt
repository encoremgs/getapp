    const appStoreUrl = "https://apps.apple.com/us/app/encore-online/id6652313770";
    const webFallbackUrl = "https://www.encoreonline.us/";

    const ua = navigator.userAgent.toLowerCase();

    if (/android/.test(ua)) {
      window.location.href = playStoreUrl;
      document.getElementById("manualLink").href = playStoreUrl;
    } else if (/iphone|ipad|ipod/.test(ua)) {
      window.location.href = appStoreUrl;
      document.getElementById("manualLink").href = appStoreUrl;
    } else {
      window.location.href = webFallbackUrl;
      document.getElementById("manualLink").href = webFallbackUrl;
